   ```cpp
   #pragma once
   #include <ArduinoJson.h>
   #include "Configuration.hpp"

   class ConfigManager {
   public:
       // Initialize: mount LittleFS, load config from flash into structs
       void Init();

       // Persistence
       void Save(bool force = false);  // Save to flash if dirty (or forced)
       bool IsDirty() const;
       void MarkDirty();
       void CheckIdleFlush(unsigned long now);  // Call from loop(), flushes after 2s idle (D-06)

       // Direct struct access for runtime reads (no flash access)
       ConfigurationData& Global();
       const ConfigurationData& Global() const;
       KeyModeData& Bank(uint8_t index);
       const KeyModeData& Bank(uint8_t index) const;
       ControlChangeData& CC(uint8_t index);
       const ControlChangeData& CC(uint8_t index) const;

       // Per-parameter mutation (marks dirty automatically, no flash write)
       void SetGlobalParam(uint8_t fieldId, uint8_t value);
       void SetBankParam(uint8_t bank, uint8_t fieldId, uint8_t value);
       void SetCCParam(uint8_t bank, uint8_t ccIndex, uint8_t channel, uint8_t id);

       // Full config serialization for SysEx dump/load
       size_t SerializeToBuffer(char* buffer, size_t maxSize);
       bool DeserializeFromBuffer(const char* buffer, size_t length);

       // Migration (called during Init)
       bool MigrateIfNeeded();

   private:
       ConfigurationData global_;
       KeyModeData banks_[BANK_AMT];
       ControlChangeData cc_[BANK_AMT];

       bool dirty_ = false;
       unsigned long lastChangeTime_ = 0;
       static constexpr unsigned long IDLE_FLUSH_MS = 2000;
       static constexpr const char* CONFIG_FILE = "/configuration_data.json";
       static constexpr uint8_t CURRENT_VERSION = 200;

       void LoadFromFlash();
       void SaveToFlash();
       bool MigrateV103ToV200(JsonDocument& doc);
       void PopulateStructsFromDoc(const JsonDocument& doc);
       void PopulateDocFromStructs(JsonDocument& doc);
   };
   ```

   **DeserializeFromBuffer(buffer, length):** Create local `JsonDocument`, `deserializeJson(doc, buffer, length)`. Check for error. Call `PopulateStructsFromDoc(doc)`. Call `MarkDirty()`. Return true on success, false on parse error.

   **Global(), Bank(index), CC(index):** Return references to `global_`, `banks_[index]`, `cc_[index]`. Const overloads too.

Implement `MigrateIfNeeded()` and `MigrateV103ToV200()` in ConfigManager.cpp per D-03 (single migration function).

**MigrateIfNeeded():**
1. Open CONFIG_FILE, deserialize into `JsonDocument`
2. Read `doc["version"] | 0`
3. If version == 103 (or any v1xx value — the current code uses 102 as default, but the config file stores whatever was saved):
   - Call `MigrateV103ToV200(doc)`
   - Save the migrated doc to flash
   - Reload structs from the migrated doc via `PopulateStructsFromDoc(doc)`
   - Return true
4. If version == 200: no migration needed, return false
5. If version is unknown (0 or other): treat as fresh install, save defaults, return false

**MigrateV103ToV200(doc):**
The v103 format stores global fields flat at root level:
```json
{"version":103, "mode":0, "sensitivity":1, "brightness":6, "midi_trs":0, ...
 "custom_scale1":[...], "custom_scale2":[...],
 "banks":[{"pal":0, "ch":1, "scale":0, ...}, ...]}
```

The v200 format nests globals under "global":
```json
{"version":200, "global":{"mode":0, "sensitivity":1, ...},
 "banks":[{"ch":1, "scale":0, ...}, ...]}
```

Migration steps:
1. Create a `JsonObject global = doc["global"].to<JsonObject>()`
2. Move each global field from root to global object:
   - `global["mode"] = doc["mode"] | 0`
   - `global["sensitivity"] = doc["sensitivity"] | 1`
   - `global["brightness"] = doc["brightness"] | 6`
   - `global["midi_trs"] = doc["midi_trs"] | 0`
   - `global["trs_type"] = doc["trs_type"] | 0`
   - `global["passthrough"] = doc["passthrough"] | 0`
   - `global["midi_ble"] = doc["midi_ble"] | 0`
   - Copy custom_scale1 array: `global["custom_scale1"] = doc["custom_scale1"]`
   - Copy custom_scale2 array: `global["custom_scale2"] = doc["custom_scale2"]`
3. Remove old root-level keys: `doc.remove("mode")`, `doc.remove("sensitivity")`, etc. for all moved keys
4. Set `doc["version"] = 200`
5. Banks array stays as-is (same structure in both versions)
6. Log: `log_d("Migrated config from v103 to v200")`
7. Return true

IMPORTANT per D-03: This is a single function handling one known legacy format. No incremental migration chain. No version-to-version stepping.
