2. Create `editor-tx/src/contexts/ConnectionContext.test.tsx`:

Test the context provider and hook:

```typescript
import { describe, it, expect } from 'vitest'
import { render, screen } from '@testing-library/react'
import { renderHook } from '@testing-library/react'
import { ConnectionProvider } from '@/contexts/ConnectionContext'
import { useConnection } from '@/hooks/useConnection'

describe('useConnection', () => {
  it('throws when used outside ConnectionProvider', () => {
    expect(() => renderHook(() => useConnection())).toThrow('must be used within ConnectionProvider')
  })
})

describe('ConnectionProvider', () => {
  it('renders children', () => {
    render(<ConnectionProvider><div>test</div></ConnectionProvider>)
    expect(screen.getByText('test')).toBeInTheDocument()
  })

  it('provides initial disconnected state', () => {
    const { result } = renderHook(() => useConnection(), {
      wrapper: ConnectionProvider,
    })
    expect(result.current.isConnected).toBe(false)
    expect(result.current.input).toBeNull()
    expect(result.current.output).toBeNull()
  })
})
```
