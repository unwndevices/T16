From firmware src/Scales.cpp (SetNoteMap algorithm):
```cpp
void SetNoteMap(uint8_t scale, uint8_t root_note, bool flip_x, bool flip_y, callback, uint8_t page) {
  int8_t *scale_notes = scales[scale];
  uint8_t note_index = 0, octave = 0;
  // find scale_length by looking for first -1
  uint8_t scale_length = 16;
  for (int i = 0; i < 16; i++) {
    if (scale_notes[i] == -1) { scale_length = i; break; }
  }
  uint8_t offset = (page * 16) % scale_length;
  note_index = note_index + offset;
  for (int i = 0; i < 16; i++) {
    if (scale_notes[note_index] == -1 || note_index == 16) {
      note_index = 0; octave++;
    }
    x = i % 4; y = i / 4;
    if (flip_x) x = 3 - x;
    if (flip_y) y = 3 - y;
    index = y * 4 + x;
    note_map[index] = (root_note + scale_notes[note_index] + (octave * 12));
    note_index++;
  }
}
```
