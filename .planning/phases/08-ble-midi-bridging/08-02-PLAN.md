    **Receiving (input or transport):**

    The SysEx receive useEffect currently uses `input.addListener('sysex', handler)`. For BLE, the transport provides `addSysexListener`. Add a parallel useEffect:

    ```typescript
    // Listen for SysEx via BLE transport (when no USB input available)
    useEffect(() => {
      if (!transport || !isConnected || input) return
      // If input exists, USB path handles SysEx â€” skip BLE listener to avoid duplicates

      const handleSysex = (data: Uint8Array) => {
        const msg = parseSysExMessage(data)
        if (!msg) return
        // Same logic as the USB sysex handler
        if (isConfigResponse(msg.cmd, msg.sub)) {
          try {
            const config = parseConfigDump(msg.payload)
            dispatch({ type: 'SET_CONFIG', payload: config })
            dispatch({ type: 'SET_DEVICE_CONFIG', payload: config })
          } catch (err) {
            console.error('Failed to parse config dump:', err)
          }
        }
        if (isParamAck(msg.cmd, msg.sub)) {
          const timestamps = pendingParamTimestamps.current
          if (timestamps.size > 0) {
            const entry = timestamps.entries().next()
            if (!entry.done) {
              const [key, sendTime] = entry.value
              const roundTrip = performance.now() - sendTime
              console.debug(`[T16 Sync] Param ${key} round-trip: ${roundTrip.toFixed(1)}ms`)
              timestamps.delete(key)
            }
          }
        }
      }

      transport.addSysexListener(handleSysex)
      return () => {
        transport.removeSysexListener(handleSysex)
      }
    }, [transport, isConnected, input])
    ```
