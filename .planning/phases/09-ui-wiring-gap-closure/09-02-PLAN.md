    4. Add import handler function inside SettingsTab:
       ```typescript
       const handleImport = (e: React.ChangeEvent<HTMLInputElement>) => {
         const file = e.target.files?.[0]
         if (!file) return
         const reader = new FileReader()
         reader.onload = () => {
           try {
             const data = JSON.parse(reader.result as string)
             const result = importConfig(data)
             if (!result.valid && result.errors) {
               setImportErrors(result.errors)
             } else {
               setImportErrors([])
             }
           } catch {
             setImportErrors(['Invalid JSON file'])
           }
         }
         reader.readAsText(file)
         // Reset input so same file can be re-imported
         e.target.value = ''
       }
       ```

    5. Add `useRef` to the imports from 'react' at the top of the file (line 1).

    6. Below the existing `.dangerZone` div in SettingsTab, add a new config management section:
       ```tsx
       <div className={styles.configSection}>
         <h3 className={styles.sectionHeading}>Configuration</h3>
         <div className={styles.configActions}>
           <Button variant="secondary" onClick={exportConfig}>
             Export Config
           </Button>
           <Button variant="secondary" onClick={() => fileInputRef.current?.click()}>
             Import Config
           </Button>
           <input
             ref={fileInputRef}
             type="file"
             accept=".topo,.json"
             onChange={handleImport}
             style={{ display: 'none' }}
           />
         </div>
         {importErrors.length > 0 && (
           <div className={styles.importErrors}>
             <p>Import failed:</p>
             <ul>
       ```
